Selections and Binding 
const arr = [1, 2, 3, 4, 5];
select('.card1')
  .selectAll('button')
  .data(arr)
  .enter()
  .append('button')
  .text(d => d);

select('.card1')
  .select('button')
  .data();// [1]

select('.card1')
  .selectAll('button')
  .data();//[1,2,3,4,5]

select('.card1')
  .selectAll('button')
  .nodes();//[button, button, ..., button]
Binding data to elements (New data enters) 
const kitties = [
  {{ '{' }} name: 'Mittens', age: 1, color: 'gray', {{ '}' }},
  {{ '{' }} name: 'Fluffy', age: 3, color: 'orange' {{ '}' }},
  {{ '{' }} name: 'Spike', age: 5, color: 'black' {{ '}' }}
  {{ '{' }} name: 'Bob', age: 6, color: 'blue' {{ '}' }}
];

const kittieSelection = select("#kitty-svg")
    .selectAll('circle')
    .data(this.kitties); // bind data to shapes

kittieSelection
    .enter() // add new elements
    .append('circle')
    .attr('fill', (kitty) => kitty.color)
    .attr('cx', (kitty) => kitty.age * 25)
    .attr('cy', 30)
    .attr('r', 40);
