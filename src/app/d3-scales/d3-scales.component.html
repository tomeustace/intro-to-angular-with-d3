Scale Linear 
 const linear = scaleLinear()
        .domain([0, 1000])
        .range([0, 500]);

    linear(0); // -> 0
    linear(500); // -> 250
    linear(1000); // -> 500
    linear.invert(500); // -> 1000

select('#linear-x')
      .selectAll('rect')
      .data([100,150,300,400,430])
      .enter()
      .append('rect')
      .attr('x', (d) => linear(d))

select('#linear-y')
      .selectAll('rect')
      .data([30,50,100,140,180])
      .enter().append('rect')
      .attr('y', (d) => linear(d))
      .attr('x', (d) => linear(d))

Scale Time 
const dates = [new Date(2022, 0, 1), new Date(2022, 0, 2)];
const timeScale = scaleTime()
  .domain(dates)
  .range([0, 500]);

timeScale(new Date(2022, 0, 1)); // 0
timeScale(new Date(2022, 0, 1, 12)); // 250
timeScale(new Date(2022, 0, 2)); // 500
timeScale.invert(0); // Sat Jan 01 2022 00:00:00 GMT-0000 (GMT)
timeScale.invert(250); // Sat Jan 01 2022 12:00:00 GMT-0000 (GMT)
timeScale.invert(500); // Sat Jan 02 2022 00:00:00 GMT-0000 (GMT)

select('#time-x')
.selectAll('circle')
.data([
  new Date(2022, 0, 1, 2),
  new Date(2022, 0, 1, 5),
  new Date(2022, 0, 1, 10)
])
.enter()
.append('circle')
.attr('r', 20)
.attr('cy', 50)
.attr('cx', (d) => timeScale(d));
